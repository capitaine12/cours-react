export default function UserCard({ user }) {
  // ðŸ”¸ Choix de la classe CSS selon le sexe
  const cardClass = user.sexe === 'f' ? 'card femme' : 'card homme';

  // ðŸ”¸ On extrait le JSX dans une fonction pour la clartÃ©
  function renderUserInfo() {
    return (
      <>
        <div className="card-img">
          <img
            src={user.imgUrl}
            alt={`${user.prenom} ${user.nom}`}
            style={{
              width: user.imgSize,
              height: user.imgSize,
              borderRadius: '50%',
            }}
          />
        </div>

        <div className="card-body">
          <h2>{user.prenom} {user.nom}</h2>
          <span className="profil">{user.profile}</span>
          <p className="desc">{user.description}</p>
        </div>
      </>
    );
  }

  // ðŸ”¸ Le return final de la carte
  return (
    <div className={cardClass}>
      {renderUserInfo()}
    </div>
  );
}
